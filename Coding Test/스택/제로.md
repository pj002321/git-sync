```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <stack>
using namespace std;

int x;
long long sum;
int main()
{
	stack<int> a;
	int k;
	cin >> k;
	for(int i=0;i<k;++i){
		cin >> x;
		if (x == 0 && !a.empty()) {
			sum -= a.top();
			a.pop();
		}
		else {
			a.push(x);
			sum+=a.top();
		}
	}
	cout << sum;
	
}
```
