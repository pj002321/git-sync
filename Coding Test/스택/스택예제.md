#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <stack>
using namespace std;

string c;
int x;

int main()
{

	int N = 0;
	cin >> N;
	stack<int> S;
	for (int i = 0; i < N; ++i) {

		cin >> c;
		if (c == "push") {
			cin >> x;
			S.push(x);
		}
		else if (c == "pop") {
			if(S.empty()) {
				cout << -1 << endl;
			} else {
				cout << S.top() << endl;
				S.pop();
			}
		}
		else if (c == "top") {
			if(S.empty()) {
				cout << -1 << endl;
			} else {
				cout << S.top() << endl;
			}
		}
		else if (c == "size") {
			cout << S.size() << endl;
		}
		else if (c == "empty") {
			cout << (S.empty() ? 1 : 0) << endl;
		}		
	}
}
