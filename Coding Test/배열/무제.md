```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main()
{

	int A, B, C;
	cin >> A >> B >> C;
	int result = A * B * C;
	string stor = to_string(result);
	vector<int> digits(10, 0);
	for (int i = 0; i < stor.size(); ++i) {
		
		digits[stor[i] - '0']++;
		
	}

	for(int i = 0; i < 10; ++i) {
		cout << digits[i] << endl;
	}
}
```
