```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;



int main()
{
	
	int n, x=0;
	int ans = 0;
	int room[2][7] = {};
	cin >> n>>x;
	int S, Y = 0;
	for (int i = 0; i < n; i++) {
		cin >> S >> Y;
		room[S][Y]++;
	}
	
	for(int i=0; i < 2; i++)
		for (int j = 1; j <= 6; j++)
			ans += (room[i][j]+x-1)/ x; // Each room can accommodate x students, so we divide by x and round up
	cout << ans;
}
```
