#include <string>
#include <vector>

using namespace std;

int solution(int a, int b, int n) {
    int answer = 0;
    int emptybottle = n;
    while(emptybottle >= a)
    {
        int exchangeCount = emptybottle / a;//6   
        int newCola = exchangeCount * b;
        answer += newCola;
        
        int remaniningBottles = emptybottle % a;//2
        emptybottle = remaniningBottles + newCola;//2+6        
    }

  
    return answer;
}
